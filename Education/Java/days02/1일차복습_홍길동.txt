package days02;
class Test{
   public static void main(String[] args){
   
     String name = "홍길동";  // 문자열 자료형   String   "문자열"
     int age;      // 정수            int       20
     age = 20;
     System.out.printf("이름 : \"%s\", 나이 : %d살입니다. ", name, age);
     
   }
}
